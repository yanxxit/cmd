# 文章模块实现方案

## 📋 需求分析

在 `/next/admin` 后台管理系统中添加一个完整的文章管理模块，包括：

- ✅ 文章列表展示（支持分页、搜索、筛选、排序）
- ✅ 文章创建（支持标题、内容、分类、标签、封面图等）
- ✅ 文章编辑（支持富文本编辑器）
- ✅ 文章查看（详情展示）
- ✅ 文章删除（支持批量删除）
- ✅ 文章发布/下架状态管理
- ✅ 使用 `@yanit/jsondb` 进行数据持久化存储

## 🏗️ 技术选型

### 1. 数据库：@yanit/jsondb (强制使用)

**选择理由：**
- ✅ 零依赖，轻量级，适合小型项目
- ✅ 使用 JSON 文件存储，无需额外数据库服务
- ✅ 支持异步操作，符合 Next.js 开发规范
- ✅ **MongoDB 风格的查询语法**，功能强大
- ✅ **支持 SQL 查询**（SELECT/INSERT/UPDATE/DELETE）
- ✅ **支持索引**，提升查询性能
- ✅ **支持内存模式**，适合测试和原型验证
- ✅ **支持 JSONB 二进制格式**，节省 30-40% 存储空间
- ✅ **支持事务**，保证数据一致性
- ✅ **支持 Schema 验证**，保证数据质量
- ✅ **Debug 模式**，方便调试和性能优化

**安装命令：**
```bash
pnpm add @yanit/jsondb
```

**基本用法：**
```typescript
import { Database } from '@yanit/jsondb';

// 初始化数据库
const db = new Database('./data/articles');
await db.open();

// 获取集合
const articles = db.collection('articles');

// 插入文档
const article = await articles.insertOne({
  title: '文章标题',
  content: '文章内容',
  status: 'published',
  createdAt: new Date().toISOString(),
});

// 查询（MongoDB 风格）
const allArticles = await articles.find().toArray();
const publishedArticles = await articles.find({ 
  status: 'published',
  createdAt: { $gte: '2024-01-01' }
}).toArray();

// 查询（SQL 风格）
const result = await articles.sql(
  'SELECT * FROM articles WHERE status = ? AND createdAt >= ? ORDER BY createdAt DESC LIMIT 10',
  ['published', '2024-01-01']
);

// 更新
await articles.updateOne(
  { _id: article._id },
  { $set: { status: 'archived' } }
);

// 删除
await articles.deleteOne({ _id: article._id });

// 关闭数据库
await db.close();
```

### 3. 图片上传

**方案选择：**
- **简单方案**：使用 Base64 编码直接存储到 JSON（适合小图片）
- **推荐方案**：上传到本地文件系统，存储文件路径

**实现方式：**
```typescript
// pages/api/upload.ts
import formidable from 'formidable';
import fs from 'fs';
import path from 'path';

export const config = {
  api: {
    bodyParser: false, // 禁用 Next.js 默认 body 解析
  },
};

export default async function handler(req, res) {
  const form = formidable({
    uploadDir: path.join(process.cwd(), 'public/uploads'),
    keepExtensions: true,
  });

  const [fields, files] = await form.parse(req);
  const uploadedFile = files.coverImage?.[0];
  
  if (uploadedFile) {
    const fileName = path.basename(uploadedFile.filepath);
    const imageUrl = `/next/uploads/${fileName}`;
    res.status(200).json({ url: imageUrl });
  }
}
```

    await articles.createIndex({ author: 1, status: 1 });
  }
  
  return db;
}

```typescript
// lib/db/articles.ts
import { Database } from '@yanit/jsondb';
import { Article, ArticleFilters, ArticleListResponse } from '../../types/article';

let dbInstance: Database | null = null;

// 初始化数据库
async function getDatabase(): Promise<Database> {
  if (!dbInstance) {
    const isDev = process.env.NODE_ENV === 'development';
    const dbPath = process.env.ARTICLES_DB_PATH || './data/articles';
    
    dbInstance = new Database(dbPath, {
      debug: isDev,
      jsonb: !isDev,
    });
    
    await dbInstance.open();
    
    // 创建索引
    const articles = dbInstance.collection('articles');
    await articles.createIndex({ status: 1 });
    await articles.createIndex({ createdAt: -1 });
    await articles.createIndex({ category: 1, status: 1 });
  }
  
  return dbInstance;
}

// 获取文章集合
async function getArticlesCollection() {
  const db = await getDatabase();
  return db.collection<Article>('articles');
}

// 创建文章
export async function createArticle(article: Omit<Article, '_id' | 'createdAt' | 'updatedAt'>) {
  const collection = await getArticlesCollection();
  const now = new Date().toISOString();
  
  const result = await collection.insertOne({
    ...article,
    viewCount: 0,
    likeCount: 0,
    createdAt: now,
    updatedAt: now,
  });
  
  return result;
}

// 获取文章列表（使用 MongoDB 风格查询）
export async function getArticles(filters: ArticleFilters, page = 1, pageSize = 10): Promise<ArticleListResponse> {
  const collection = await getArticlesCollection();
  
  // 构建查询条件
  const query: any = {};
  
  if (filters.status) {
    query.status = filters.status;
  }
  
  if (filters.category) {
    query.category = filters.category;
  }
  
  if (filters.keyword) {
    query.$or = [
      { title: { $regex: filters.keyword, $options: 'i' } },
      { summary: { $regex: filters.keyword, $options: 'i' } }
    ];
  }
  
  if (filters.startDate || filters.endDate) {
    query.createdAt = {};
    if (filters.startDate) query.createdAt.$gte = filters.startDate;
    if (filters.endDate) query.createdAt.$lte = filters.endDate;
  }
  
  // 计算总数
  const total = await collection.countDocuments(query);
  
  // 查询数据
  const articles = await collection.find(query)
    .sort({ createdAt: -1 })
    .skip((page - 1) * pageSize)
    .limit(pageSize)
    .toArray();
  
  return {
    data: articles,
    total,
    page,
    pageSize,
  };
}

// 获取文章列表（使用 SQL 查询）
export async function getArticlesSQL(filters: ArticleFilters, page = 1, pageSize = 10): Promise<ArticleListResponse> {
  const collection = await getArticlesCollection();
  
  // 构建 WHERE 条件
  const conditions: string[] = [];
  const params: any[] = [];
  
  if (filters.status) {
    conditions.push('status = ?');
    params.push(filters.status);
  }
  
  if (filters.category) {
    conditions.push('category = ?');
    params.push(filters.category);
  }
  
  if (filters.keyword) {
    conditions.push('(title LIKE ? OR summary LIKE ?)');
    const keyword = `%${filters.keyword}%`;
    params.push(keyword, keyword);
  }
  
  if (filters.startDate) {
    conditions.push('createdAt >= ?');
    params.push(filters.startDate);
  }
  
  if (filters.endDate) {
    conditions.push('createdAt <= ?');
    params.push(filters.endDate);
  }
  
  const whereClause = conditions.length > 0 ? `WHERE ${conditions.join(' AND ')}` : '';
  
  // 查询总数
  const countQuery = `SELECT COUNT(*) as total FROM articles ${whereClause}`;
  const countResult = await collection.sql(countQuery, params);
  const total = countResult.data[0]?.total || 0;
  
  // 查询数据
  const dataQuery = `
    SELECT * FROM articles
    ${whereClause}
    ORDER BY createdAt DESC
    LIMIT ? OFFSET ?
  `;
  const dataParams = [...params, pageSize, (page - 1) * pageSize];
  const result = await collection.sql(dataQuery, dataParams);
  
  return {
    data: result.data,
    total,
    page,
    pageSize,
  };
}

// 获取单篇文章
export async function getArticleById(id: string): Promise<Article | null> {
  const collection = await getArticlesCollection();
  const article = await collection.findOne({ _id: id });
  return article || null;
}

// 更新文章
export async function updateArticle(id: string, updates: Partial<Article>): Promise<Article | null> {
  const collection = await getArticlesCollection();
  
  await collection.updateOne(
    { _id: id },
    { 
      $set: {
        ...updates,
        updatedAt: new Date().toISOString(),
      }
    }
  );
  
  return getArticleById(id);
}

// 删除文章
export async function deleteArticle(id: string): Promise<void> {
  const collection = await getArticlesCollection();
  await collection.deleteOne({ _id: id });
}

// 批量删除文章
export async function deleteArticles(ids: string[]): Promise<void> {
  const collection = await getArticlesCollection();
  await collection.deleteMany({ _id: { $in: ids } });
}

// 统计文章数量
export async function countArticles(filters?: ArticleFilters): Promise<number> {
  const collection = await getArticlesCollection();
  
  if (!filters) {
    return collection.countDocuments();
  }
  
  const query: any = {};
  if (filters.status) query.status = filters.status;
  if (filters.category) query.category = filters.category;
  
  return collection.countDocuments(query);
}

// 获取统计信息（使用聚合）
export async function getArticleStats() {
  const collection = await getArticlesCollection();
  
  const stats = await collection.aggregate([
    {
      $group: {
        _id: '$status',
        count: { $sum: 1 },
        totalViews: { $sum: '$viewCount' },
        totalLikes: { $sum: '$likeCount' }
      }
    }
  ]);
  
  return stats;
}
```

```typescript
// pages/api/articles/index.ts
import type { NextApiRequest, NextApiResponse } from 'next';
import {
  getArticles,
  createArticle,
  countArticles,
} from '../../../lib/db/articles';
import { Article, ArticleFilters } from '../../../types/article';

export default async function handler(
  req: NextApiRequest,
  res: NextApiResponse
) {
  const { method } = req;
  
  try {
    switch (method) {
      case 'GET': {
        // 获取文章列表
        const {
          page = '1',
          pageSize = '10',
          keyword,
          status,
          category,
          tag,
          author,
          sortBy = 'createdAt',
          sortOrder = 'desc',
          startDate,
          endDate,
        } = req.query;
        
        const filters: ArticleFilters = {
          keyword: keyword as string,
          status: status as Article['status'],
          category: category as string,
          tag: tag as string,
          author: author as string,
          startDate: startDate as string,
          endDate: endDate as string,
        };
        
        const result = await getArticles(
          filters,
          parseInt(page as string),
          parseInt(pageSize as string)
        );
        
        res.status(200).json(result);
        break;
      }
      
      case 'POST': {
        // 创建文章
        const article = req.body;
        
        // 验证必填字段
        if (!article.title || !article.content || !article.author) {
          return res.status(400).json({
            error: '缺少必填字段',
            message: 'title, content, author 是必填的',
          });
        }
        
        const created = await createArticle(article);
        res.status(201).json(created);
        break;
      }
      
      default:
        res.setHeader('Allow', ['GET', 'POST']);
        res.status(405).end(`Method ${method} Not Allowed`);
    }
  } catch (error: any) {
    console.error('API Error:', error);
    res.status(500).json({
      error: '服务器错误',
      message: error.message,
    });
  }
}
```

```typescript
// pages/admin/articles/index.tsx
import React, { useState, useEffect } from 'react';
import {

  Card,
  Tag,
  Popconfirm,
  message,
  Typography,
  Row,
  Col,
  Statistic,
} from 'antd';
import {
  SearchOutlined,
  PlusOutlined,
  EditOutlined,
  DeleteOutlined,
  EyeOutlined,
} from '@ant-design/icons';
import { AdminLayout } from '../../../components/admin/layout/AdminLayout';
import { Article } from '../../../types/article';
import { request } from '../../../lib/request';
import Link from 'next/link';

const { Title } = Typography;

const statusMap = {
  draft: { color: 'gray', text: '草稿' },
  published: { color: 'green', text: '已发布' },
  archived: { color: 'orange', text: '已归档' },
};

export default function ArticleManagement() {
  const [articles, setArticles] = useState<Article[]>([]);
  const [loading, setLoading] = useState(false);
  const [total, setTotal] = useState(0);
  const [pagination, setPagination] = useState({
    current: 1,
    pageSize: 10,
  });
  
  const [filters, setFilters] = useState({
    keyword: '',
    status: '',
    category: '',
  });
  
  // 统计数据
  const [stats, setStats] = useState({
    total: 0,
    published: 0,
    draft: 0,
    archived: 0,
  });
  
  // 加载文章列表
  const loadArticles = async () => {
    setLoading(true);
    try {
      const params = {
        page: pagination.current,
        pageSize: pagination.pageSize,
        ...filters,
      };
      
      const data = await request('/articles', { params });
      setArticles(data.data);
      setTotal(data.total);
    } catch (error) {
      message.error('加载文章列表失败');
    } finally {
      setLoading(false);
    }
  };
  
  // 加载统计数据
  const loadStats = async () => {
    try {
      const data = await request('/articles/stats');
      setStats(data);
    } catch (error) {
      console.error('加载统计失败', error);
    }
  };
  
  useEffect(() => {
    loadArticles();
    loadStats();
  }, [pagination.current, pagination.pageSize]);
  
  // 删除文章
  const handleDelete = async (id: string) => {
    try {
      await request(`/articles/${id}`, { method: 'DELETE' });

      loadStats();

      message.error('删除失败');

            <Typography.Text type="secondary" ellipsis style={{ fontSize: 12 }}>
              {record.summary}
            </Typography.Text>
          )}
        </div>
      ),
    },
    {
      title: '作者',
      dataIndex: 'author',
      key: 'author',
      width: 100,
    },
    {
      title: '分类',
      dataIndex: 'category',
      key: 'category',
      width: 100,
      render: (category?: string) => category || '-',
    },
    {
      title: '状态',
      dataIndex: 'status',
      key: 'status',
      width: 100,
      render: (status: keyof typeof statusMap) => {
        const config = statusMap[status];
        return <Tag color={config.color}>{config.text}</Tag>;
      },

    {
      title: '阅读量',
      dataIndex: 'viewCount',
      key: 'viewCount',
      width: 80,
      sorter: (a, b) => a.viewCount - b.viewCount,
    },

      dataIndex: 'likeCount',

        />
      </Card>
    </AdminLayout>
  );
}
```
